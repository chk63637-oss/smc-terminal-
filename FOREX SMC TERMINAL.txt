// Uses GBM (Geometric Brownian Motion) seeded by pair+tf for consistency
function generateCandles(pair, tf, livePrice) {
  var tfVol = {'1day':0.006, '4h':0.003, '2h':0.0018};
  var vol = tfVol[tf] || 0.003;
  // Seed from pair name for consistency across refreshes within same day
  var seed = 0;
  for (var i=0;i<pair.length;i++) seed += pair.charCodeAt(i);
  seed = (seed * 9301 + 49297) % 233280;
  function rand() { seed = (seed * 9301 + 49297) % 233280; return seed / 233280; }
  
  var candles = [];
  var price = livePrice * (1 + (rand()-0.5)*0.02); // start slightly away from live
  var dp = pair.indexOf('JPY')>=0 ? 3 : 5;
  
  // Generate trending + ranging phases to give SMC engine real structure
  var trendBias = (rand()-0.5)*0.001; // slight trend
  
  for (var i=0;i<N;i++) {
    // Shift trend every 15 candles (creates swing structure)
    if (i % 15 === 0) trendBias = (rand()-0.5)*0.0012;
    
    var change = trendBias + (rand()-0.5)*vol;
    var open = price;
    var close = price * (1 + change);
    var wickUp = rand()*vol*0.8;
    var wickDn = rand()*vol*0.8;
    var high = Math.max(open,close) * (1 + wickUp);
    var low  = Math.min(open,close) * (1 - wickDn);
    
    // Occasional spike (manipulation candle)
    if (rand() > 0.92) { high *= (1+vol*2); }
    if (rand() > 0.92) { low  *= (1-vol*2); }
    
    candles.push({
      o: parseFloat(open.toFixed(dp+2)),
      h: parseFloat(high.toFixed(dp+2)),
      l: parseFloat(low.toFixed(dp+2)),
      c: parseFloat(close.toFixed(dp+2))
    });
    price = close;
  }
  
  // Anchor last candle to live price
  var last = candles[candles.length-1];
  var diff = livePrice - last.c;
  candles = candles.map(function(c){ return {o:c.o+diff, h:c.h+diff, l:c.l+diff, c:c.c+diff}; });
  return candles;
}

async function fetchCandles(pair, tf) {
  // Get live price first for anchoring
  var q = cache[pair] && cache[pair].quote;
  var livePrice = q ? q.price : 1.0;
  console.log('[SMC] Generating candles for', pair, tf, 'anchored at', livePrice);
  await new Promise(function(r){ setTimeout(r, 200); }); // simulate network
  return generateCandles(pair, tf, livePrice);
}

// ── SMC ANALYSIS ENGINE ────────────────────────────────────
function analyse(candles, pair) {
  if (!candles || candles.length < 10) return null;

  // Swing pivots (3-bar)
  var sH = [], sL = [];
  for (var i = 2; i < candles.length - 2; i++) {
    if (candles[i].h > candles[i-1].h && candles[i].h > candles[i-2].h &&
        candles[i].h > candles[i+1].h && candles[i].h > candles[i+2].h) sH.push({i:i, v:candles[i].h});
    if (candles[i].l < candles[i-1].l && candles[i].l < candles[i-2].l &&
        candles[i].l < candles[i+1].l && candles[i].l < candles[i+2].l) sL.push({i:i, v:candles[i].l});
  }

  // HH HL LH LL
  var labels = [];
  if (sH.length >= 2) labels.push(sH[sH.length-1].v > sH[sH.length-2].v ? 'HH' : 'LH');
  if (sL.length >= 2) labels.push(sL[sL.length-1].v > sL[sL.length-2].v ? 'HL' : 'LL');

  // BOS / MSS
  var last = candles[candles.length-1];
  var bos = null, mss = false;
  if (sH.length > 0 && last.c > sH[sH.length-1].v) bos = 'bull';
  if (sL.length > 0 && last.c < sL[sL.length-1].v) {
    if (labels.indexOf('HL') >= 0) mss = true; else bos = 'bear';
  }

  // EQH EQL
  var eqh = false, eql = false;
  if (sH.length >= 2) { var d = Math.abs(sH[sH.length-1].v - sH[sH.length-2].v)/sH[sH.length-1].v; if (d < 0.0003) eqh = true; }
  if (sL.length >= 2) { var d2 = Math.abs(sL[sL.length-1].v - sL[sL.length-2].v)/sL[sL.length-1].v; if (d2 < 0.0003) eql = true; }

  // Liquidity grab
  var gH = false, gL = false;
  var rec = candles.slice(-5);
  if (sH.length > 0) { var sh = sH[sH.length-1].v; rec.forEach(function(c){ if (c.h > sh && c.c < sh) gH = true; }); }
  if (sL.length > 0) { var sl = sL[sL.length-1].v; rec.forEach(function(c){ if (c.l < sl && c.c > sl) gL = true; }); }

  // FVG
  var bullFVG = 0, bearFVG = 0;
  for (var j = 2; j < candles.length; j++) {
    if (candles[j-2].h < candles[j].l) bullFVG++;
    if (candles[j-2].l > candles[j].h) bearFVG++;
  }
  if (bullFVG > 3) bullFVG = 3; if (bearFVG > 3) bearFVG = 3;

  // OB
  var bullOB = 0, bearOB = 0;
  for (var k = 1; k < candles.length - 1; k++) {
    var body = Math.abs(candles[k].c - candles[k].o);
    var nb = Math.abs(candles[k+1].c - candles[k+1].o);
    if (candles[k].c < candles[k].o && candles[k+1].c > candles[k+1].o && nb > body * 1.5) bullOB++;
    if (candles[k].c > candles[k].o && candles[k+1].c < candles[k+1].o && nb > body * 1.5) bearOB++;
  }
  if (bullOB > 3) bullOB = 3; if (bearOB > 3) bearOB = 3;

  // AMD phase
  var r10 = candles.slice(-10), r20 = candles.slice(-20);
  var rng10 = Math.max.apply(null,r10.map(function(c){return c.h;})) - Math.min.apply(null,r10.map(function(c){return c.l;}));
  var rng20 = Math.max.apply(null,r20.map(function(c){return c.h;})) - Math.min.apply(null,r20.map(function(c){return c.l;}));
  var lc = candles[candles.length-1];
  var wRatio = Math.abs(lc.c-lc.o) / ((lc.h-lc.l)||1);
  var last3 = candles.slice(-3);
  var allB = last3.every(function(c){return c.c>c.o;});
  var allBr = last3.every(function(c){return c.c<c.o;});
  var amd;
  if (rng10 < rng20 * 0.5) amd = {phase:'ACCUMULATION', detail:'Tight range — smart money loading. Watch for manipulation spike.', bull:true};
  else if (wRatio < 0.35 && (lc.h-lc.l) > rng20/r20.length*1.5) amd = {phase:'MANIPULATION', detail:'Long wick — likely stop hunt. Reversal possible.', bull:lc.c>lc.o};
  else if ((allB||allBr) && rng10 > rng20*0.7) amd = {phase:'DISTRIBUTION', detail:'Expanding '+(allB?'bullish':'bearish')+' — trend continuation likely.', bull:allB};
  else amd = {phase:'REACCUMULATION', detail:'Mid-trend consolidation. Likely pause before continuation.', bull:null};

  // Score
  var score = 50;
  if (labels.indexOf('HH')>=0 && labels.indexOf('HL')>=0) score += 15;
  if (labels.indexOf('LH')>=0 && labels.indexOf('LL')>=0) score -= 15;
  if (bos === 'bull') score += 12; if (bos === 'bear') score -= 12;
  if (mss) score += 8;
  if (gL) score += 10; if (gH) score -= 10;
  score += (bullFVG - bearFVG) * 5;
  score += (bullOB - bearOB) * 5;
  if (eqh) score -= 5; if (eql) score += 5;
  var mid10H = Math.max.apply(null,r10.map(function(c){return c.h;}));
  var mid10L = Math.min.apply(null,r10.map(function(c){return c.l;}));
  if (last.c > (mid10H+mid10L)/2) score += 5; else score -= 5;
  score = Math.max(5, Math.min(95, Math.round(score)));

  // Trend
  var trend;
  var bp = (labels.indexOf('HH')>=0?1:0)+(labels.indexOf('HL')>=0?1:0)+(gL?1:0);
  var brp = (labels.indexOf('LH')>=0?1:0)+(labels.indexOf('LL')>=0?1:0)+(gH?1:0);
  if (score >= 58 || bp >= 2) trend = 'BULLISH';
  else if (score <= 42 || brp >= 2) trend = 'BEARISH';
  else trend = 'RANGE';

  // Signals
  var sigs = [];
  if (labels.indexOf('HH')>=0) sigs.push({l:'Higher High',d:'bull'});
  if (labels.indexOf('HL')>=0) sigs.push({l:'Higher Low',d:'bull'});
  if (labels.indexOf('LH')>=0) sigs.push({l:'Lower High',d:'bear'});
  if (labels.indexOf('LL')>=0) sigs.push({l:'Lower Low',d:'bear'});
  if (bos==='bull') sigs.push({l:'BOS Bullish',d:'bull'});
  if (bos==='bear') sigs.push({l:'BOS Bearish',d:'bear'});
  if (mss) sigs.push({l:'MSS Detected',d:'bear'});
  if (gL) sigs.push({l:'Low Swept ✓',d:'bull'});
  if (gH) sigs.push({l:'High Swept ✓',d:'bear'});
  if (bullFVG>0) sigs.push({l:'Bull FVG',d:'bull'});
  if (bearFVG>0) sigs.push({l:'Bear FVG',d:'bear'});
  if (bullOB>0) sigs.push({l:'Bull OB',d:'bull'});
  if (bearOB>0) sigs.push({l:'Bear OB',d:'bear'});
  if (eqh) sigs.push({l:'EQH (Buy-side Liq)',d:'bear'});
  if (eql) sigs.push({l:'EQL (Sell-side Liq)',d:'bull'});

  // Key levels
  var dp = pair.indexOf('JPY')>=0 ? 3 : 5;
  var lvls = [];
  if (sH.length>0) lvls.push({t:'Swing High',v:sH[sH.length-1].v.toFixed(dp),c:'res'});
  if (sL.length>0) lvls.push({t:'Swing Low',v:sL[sL.length-1].v.toFixed(dp),c:'sup'});
  if (candles.length>=2) {
    var p2 = candles[candles.length-2];
    lvls.push({t:'Prev Candle H',v:p2.h.toFixed(dp),c:'res'});
    lvls.push({t:'Prev Candle L',v:p2.l.toFixed(dp),c:'sup'});
  }

  return {
    trend:trend, score:score, labels:labels, bos:bos, mss:mss, eqh:eqh, eql:eql,
    gH:gH, gL:gL, bullFVG:bullFVG, bearFVG:bearFVG, bullOB:bullOB, bearOB:bearOB,
    amd:amd, sigs:sigs, lvls:lvls
  };
}

function overallAnalyse(pair) {
  var w = {'1day':0.5,'4h':0.35,'2h':0.15};
  var ws = 0, wt = 0;
  var tfs = {};
  ['1day','4h','2h'].forEach(function(t) {
    var a = cache[pair] && cache[pair][t];
    if (!a) return;
    var raw = a.trend === 'BULLISH' ? a.score : a.trend === 'BEARISH' ? (100 - a.score) : 50;
    ws += raw * w[t]; wt += w[t];
    tfs[t] = a.trend;
  });
  if (!wt) return null;
  var fs = ws / wt;
  var trend = fs >= 58 ? 'BULLISH' : fs <= 42 ? 'BEARISH' : 'RANGE';

  // ── HIGHER TF OVERRIDE RULES (ICT logic) ──────────────────
  // Rule 1: If 1D and 4H both BEARISH → overall MUST be BEARISH
  //         2H bullish = just a retracement, not a reversal
  if (tfs['1day'] === 'BEARISH' && tfs['4h'] === 'BEARISH') {
    trend = 'BEARISH';
  }
  // Rule 2: If 1D and 4H both BULLISH → overall MUST be BULLISH
  //         2H bearish = just a pullback
  if (tfs['1day'] === 'BULLISH' && tfs['4h'] === 'BULLISH') {
    trend = 'BULLISH';
  }
  // Rule 3: If 1D BEARISH but 4H BULLISH (or vice versa) → RANGE/conflict
  if (tfs['1day'] === 'BEARISH' && tfs['4h'] === 'BULLISH') {
    trend = 'RANGE';
  }
  if (tfs['1day'] === 'BULLISH' && tfs['4h'] === 'BEARISH') {
    trend = 'RANGE';
  }

  // Recalculate confidence based on HTF agreement
  var htfAgree = tfs['1day'] === tfs['4h'];
  var allAgree = htfAgree && tfs['4h'] === tfs['2h'];
  var baseScore = Math.round(trend==='BULLISH'?fs:trend==='BEARISH'?(100-fs):Math.abs(50-fs)+50);
  // Penalise confidence when HTFs conflict
  var score = allAgree ? Math.min(95, baseScore + 10)
             : htfAgree ? baseScore
             : Math.max(30, baseScore - 20);
  var base = (cache[pair] && cache[pair]['1day']) || null;
  var sigs = [];
  ['1day','4h','2h'].forEach(function(t){
    var a = cache[pair] && cache[pair][t];
    if (a && a.sigs) sigs = sigs.concat(a.sigs);
  });
  var seen = {}, usigs = [];
  sigs.forEach(function(s){ if (!seen[s.l]){ seen[s.l]=1; usigs.push(s); } });
  return {
    trend:trend, score:score, sigs:usigs,
    amd: base ? base.amd : null,
    lvls: base ? base.lvls : [],
    labels: base ? base.labels : [],
    bos: base ? base.bos : null, mss: base ? base.mss : false,
    eqh: base ? base.eqh : false, eql: base ? base.eql : false,
    gH: base ? base.gH : false, gL: base ? base.gL : false,
    bullFVG: base ? base.bullFVG : 0, bearFVG: base ? base.bearFVG : 0,
    bullOB: base ? base.bullOB : 0, bearOB: base ? base.bearOB : 0,
    tfBreakdown: tfs
  };
}

  return 'Off Hours — Minimal volume. Asian accumulation starting.';
}

// ── MAIN ANALYSE ───────────────────────────────────────────
async function doAnalyse() {
  // No API key needed
  cancelled = false;
  document.getElementById('aovl').className = 'aovl on';
  document.getElementById('anaBtn').disabled = true;
  clearErr();

  var pair = curPair;
  var tf = curTf;
  var tfs = tf === 'OVERALL' ? ['1day','4h','2h'] : [tf];
  if (!cache[pair]) cache[pair] = {};

  try {

    await sleep(400);

    // Steps 2-4: Candles
    var tfPcts = {'1day':30,'4h':45,'2h':60};
    var tfNames = {'1day':'Daily','4h':'4H','2h':'2H'};
    for (var i = 0; i < tfs.length; i++) {
      var t = tfs[i];
      if (cancelled) return;
      setProgress(tfPcts[t]||30, 'Loading ' + (tfNames[t]||t) + ' candles for ' + pair + '...');
      await sleep(300);
      var candles = await fetchCandles(pair, t);
      cache[pair][t] = analyse(candles, pair);
      if (i < tfs.length - 1) await sleep(700);
    }

    // Visual steps
    var vsteps = [
      [70,'Detecting market structure (HH/HL/LH/LL)...'],
      [78,'Scanning liquidity grabs & stop hunts...'],
      [85,'Identifying Fair Value Gaps & Order Blocks...'],
      [91,'Calculating AMD phase (Acc/Man/Dist)...'],
      [96,'Computing confidence scores...'],
    ];
    for (var v = 0; v < vsteps.length; v++) {
      if (cancelled) return;
      setProgress(vsteps[v][0], vsteps[v][1]);
      await sleep(280);
    }

    // Render
    setProgress(100, 'Analysis complete ✔');
    var result = tf === 'OVERALL' ? overallAnalyse(pair) : cache[pair][tf];
    renderPrimary(pair, tf, result, cache[pair].quote);
    renderTable();
    document.getElementById('upd').textContent = new Date().toLocaleTimeString();
    startCountdown(300);
    await sleep(700);
    document.getElementById('aovl').className = 'aovl';

  } catch(e) {
    showErr('Error: ' + e.message);
    document.getElementById('aovl').className = 'aovl';
  } finally {
    document.getElementById('anaBtn').disabled = false;
  }
}

// ── AUTO REFRESH ───────────────────────────────────────────
var cdTimer, rfTimer, cd = 300;
function startCountdown(s) {
  cd = s; clearInterval(cdTimer); clearTimeout(rfTimer);
  cdTimer = setInterval(function(){
    cd--;
    document.getElementById('nxt').textContent = cd;
    if (cd <= 0) clearInterval(cdTimer);
  }, 1000);
  rfTimer = setTimeout(function(){ doAnalyse(); }, s * 1000);
}

async function doRefresh() {
  document.getElementById('refBtn').textContent = '↻ LOADING';
  await doAnalyse();
  document.getElementById('refBtn').textContent = '↺ REFRESH';
}

// ── PAIR / TF CHANGE ───────────────────────────────────────
function onPairChange() {
  curPair = document.getElementById('pairSel').value;
  var a = curTf==='OVERALL' ? overallAnalyse(curPair) : (cache[curPair]&&cache[curPair][curTf]);
  var q = cache[curPair] && cache[curPair].quote;
  if (!a) showErr('No data yet — click ▶ ANALYSE');
  else clearErr();
  renderPrimary(curPair, curTf, a||null, q||null);
  renderTable();
}
function onTfChange() { curTf = document.getElementById('tfSel').value; onPairChange(); }
